Requete 1

Liste des musiques

SELECT ?musique ?name ?director ?release_date WHERE { ?musique wdt:P31 wd:Q11424. ?musique wdt:P495 wd:Q30. ?musique rdfs:label ?name. OPTIONAL{?musique wdt:P57 ?director.} OPTIONAL{?musique wdt:P577 ?release_date.} } Limit 1000
